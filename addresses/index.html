def get_if_rebalance_config_public_key(
    program_id: Pubkey,
    in_market_index: int,
    out_market_index: int,
) -> Pubkey:
    return Pubkey.find_program_address(
        [
            b"if_rebalance_config",
            int_to_le_bytes(in_market_index),
            int_to_le_bytes(out_market_index),
        ],
        program_id,
    )[0]
